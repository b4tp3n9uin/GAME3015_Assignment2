133XMFLOAT4X4 SceneNode::getWorldTransform() const
134{
135 XMFLOAT4X4 transform = MathHelper::Identity4x4();
136 XMMATRIX T = XMLoadFloat4x4(&transform);
137
138 for (const SceneNode* node = this; node != nullptr; node = node->mParent)
139 {
140 XMMATRIX Tp = XMLoadFloat4x4(&node->getTransform());
141 T = Tp * T;
142 }
143 XMStoreFloat4x4(&transform, T);
